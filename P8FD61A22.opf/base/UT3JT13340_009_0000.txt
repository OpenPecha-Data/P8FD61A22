ཟབ་ཆོས་ཞི་ཁྲོ་དགོངས་པ་རང་གྲོལ་གྱི་བླ་མ་བརྒྱུད་རིམ་གྱི་བསྐང་བ་འབྲེལ་ཚད་རང་གྲོལ་ཞེས་བྱ་བ་བཞུགས་སོ། རོལ་མོའི་དབྱངས་དང་བཅས་པས་འདི་སྐད་དོ། ཨོཾ་ཨཱཿཧཱུྃ། ངོ་བོ་ཀ་དག་ཆོས་སྐུའི་ཕོ་བྲང་ནས། །ཆོས་སྐུ་ཀུན་བཟང་ལྷ་ཚོགས་ཐུགས་དམ་བསྐང༌། །རང་བཞིན་ལྷུན་གྲུབ་ལོངས་སྐུའི་གཞལ་ཡས་ནས། །རྡོར་འཆང་ལོངས་སྐུའི་ལྷ་ཚོགས་ཐུགས་དམ་བསྐང༌། །ཐུགས་རྗེ་ཕྱོགས་མེད་སྤྲུལ་སྐུའི་ཕོ་བྲང་ནས། །རྡོར་སེམས་སྤྲུལ་སྐུའི་ལྷ་ཚོགས༴ མཁའ་སྤྱོད་བདེ་ལྡན་གནས་མཆོག་ཆེན་པོ་ནས། །གུ་རུ་དགའ་རབ་རྡོ་རྗེའི༴ ལྷུན་གྲུབ་ཆོས་དབྱིངས་གནས་མཆོག་དམ་པ་ནས། །རིག་འཛིན་ཤྲཱི་སེངྷའི༴ ཟངས་མདོག་དཔལ་རིའི་གནས་མཆོག་དམ་པ་ནས། །སློབ་དཔོན་པདྨའི་ལྷ་ཚོགས༴ ཨོ་རྒྱན་འདུས་པའི་གནས་མཆོག་ཆེན་པོ་ནས། །མཚོ་རྒྱལ་ཌཱ་ཀིའི་ལྷ་ཚོགས༴ སྒམ་པོ་དཔལ་རིའི་གནས་མཆོག་ཆེན་པོ་ནས། །ཆོས་རྒྱལ་ཀརྨ་གླིང་པའི༴ གར་བཞུགས་བདེ་ཆེན་ཆོས་ཀྱི་ཕོ་བྲང་ནས། །ཐུགས་སྲས་ཉི་ཟླ་ཆོས་རྗེའི༴ འགྲོ་འདུལ་ཐབས་མཁས་ཐུགས་རྗེའི་ཀློང་ཡངས་ནས། །སྲས་མཆོག་ཉི་ཟླ་འོད་ཟེར༴ སྒྲུབ་གནས་རིན་ཆེན་གླིང་གི་ཕོ་བྲང་ནས། །ནམ་མཁའ་ཆོས་ཀྱི་རྒྱལ་མཚན༴ སྨིན་གྲོལ་བདེ་ལ་བཀོད་པའི་ཕོ་བྲང་ནས། །མཁས་གྲུབ་ལྷ་དབང་མཚན་ཅན༴ བསྐྱེད་རྫོགས་བརྟན་པ་ཐོབ་པའི་ཕོ་བྲང་ནས། །ཏིང་འཛིན་ཆོས་ཀྱི་རྒྱལ་པོའི༴ ཟག་མེད་བདེ་བ་ཆེན་པོའི་གཞལ་ཡས་ནས། །སྤྲུལ་སྐུ་ཀརྨ་གུ་རུའི༴ ཕྱོགས་མེད་འགྲོ་བ་འདུལ་བའི་ཕོ་བྲང་ནས། །གར་དབང་པདྨ་རྒྱལ་མཚན༴ གང་འདུལ་འགྲོ་དོན་མཛད་པའི་གཞལ་ཡས་ནས། །མཚན་ལྡན་པདྨ་མ་ཏིའི༴ རིམ་གཉིས་རྩལ་རྫོགས་བྱིན་རླབས་ཕོ་བྲང་ནས། །སྤྲུལ་སྐུ་ཐུགས་རྗེའི་འོད་ཟེར༴ སྣང་སྲིད་སྒྱུ་མ་རང་གྲོལ་གཞལ་ཡས་ནས། །འཁྲུལ་ཞིག་བུདྡྷ་མངྒའི༴ ཟབ་གསང་སྨིན་གྲོལ་མཆོག་གི་ཕོ་བྲང་ནས། །རིག་འཛིན་ཀུན་དགའ་གྲགས་པའི༴ ཕྱོགས་མེད་འགྲོ་འདུལ་ཐུགས་རྗེའི་ཕོ་བྲང་ནས། །གར་དབང་ཀུན་དགའ་བསྟན་འཛིན༴

བདེ་ཆེན་དགའ་བཞི་དབང་རྫོགས་གཞལ་ཡས་ནས། །རིག་འཛིན་སྦས་པའི་རྣལ་འབྱོར༴ བསམ་ཡས་ཡོན་ཏན་ཀུན་རྫོགས་ཕོ་བྲང་ནས། །མཁས་གྲུབ་ཀརྨ་ཆགས་མེད༴ གང་འདུལ་འགྲོ་དོན་མཛད་པའི་ཕོ་བྲང་ནས། །རིག་འཛིན་ཨ་ནནྟ་ཡི༴ མདོ་སྔགས་ཟབ་གསང་རྣམ་རོལ་གཞལ་ཡས་ནས། །འགྲན་བྲལ་སྤྲུལ་སྐུ་མཆོག་གི༴ བསྐྱེད་རྫོགས་ཡོན་ཏན་མཐར་ཕྱིན་ཕོ་བྲང་ནས། །མཁས་མཆོག་བུདྡྷའི་མཚན་ཅན༴ སྤྱི་གཙུག་ཉི་ཟླ་པདྨའི་གདན་སྟེང་ནས། །དྲིན་ཆེན་རྩ་བའི་བླ་མའི༴ རང་ལུས་རྒྱལ་བའི་དཀྱིལ་འཁོར་ཆེན་པོ་ནས། །ཡི་དམ་ཞི་ཁྲོའི་ལྷ་ཚོགས༴ བཀའ་དང་དམ་ཚིག་གཉན་པོའི་ཕོ་བྲང་ནས། །བཀའ་སྲུང་དམ་ཅན་རྒྱ་མཚོའི༴ པདྨས་བཀའ་བསྒོས་གཏེར་གྱི་གཉེར་གཏད་པའི། །གཏེར་བདག་གཏེར་སྲུང་རྣམས་ཀྱི༴ ཐུགས་དམ་གཉན་པོ་བསྐང་བའི་དམ་རྫས་ནི། །མཆོག་ཆེན་མཆོག་ཕྲན་མཚན་ལྡན་དང༌། །ཟ་ལེ་འབྲི་དང་ཤེལ་སྒོང་ལུག །ཙ་ར་མཁའ་ལྡིང་མདངས་ཀྱིས་བསྐང༌། །བདུད་རྩི་སྨན་གྱི་མཆོད་པ་དང༌། །རཀྟ་དམར་གྱི་ཨརྒྷཾ་དང༌། །རྨད་དུ་བྱུང་བའི་གཏོར་མ་དང༌། །རྣལ་འབྱོར་ཚོགས་ཀྱི་ལོངས་སྤྱོད་དང༌། །ཤ་ཆེན་ལང་ལིང་སྣ་ཚོགས་དང༌། །སྣང་གསལ་ཞུན་གྱི་མར་མེ་དང༌། །ཤ་ཆེན་གསུར་གྱི་དུད་པ་དང༌། །ཐོད་པ་ལྟག་སྤྲོད་རྔ་སྒྲ་དང༌། །རྐང་ཆེན་དུང་གི་གླིང་བུ་དང༌། །ཐུན་མོང་མཆོད་པ་སྣ་ཚོགས་ཀྱིས། །བླ་མ་ཡི་དམ་མཁའ་འགྲོའི་ཐུགས་དམ་བསྐང༌། །བཀའ་སྲུང་དམ་ཅན་རྣམས་ཀྱི་ཐུགས་དམ་བསྐང༌། །ཁྱེད་ཀྱི་ཐུགས་དམ་བསྐང་གྱུར་ཅིག །ཁྱེད་ཀྱི་ཐུགས་དམ་བསྐང་ནས་ཀྱང༌། །བདག་གི་གཡར་དམ་སོར་སྩོལ་ཅིག །གཡར་དམ་སྐོང་ལ་རྐྱེན་དབང་ཟློག །མཆོག་དང་ཐུན་མོང་དངོས་གྲུབ་སྩོལ། །ཕྲིན་ལས་རྣམ་བཞི་འགྲུབ་པར་མཛོད། །ཅེས་བླ་མ་བརྒྱུད་རིམ་གྱི་བསྐང་བ་རྫོགས་སོ། །
